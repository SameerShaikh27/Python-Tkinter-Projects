from tkinter import *


class LoanCalculator:

    def __init__(self):

        root = Tk()
        root.title("Loan Calculator")
        root.geometry("700x500")

        Label(root, text="Loan Calculator", font=("Cooper", "30", "bold", "underline"), pady=30, padx=20).grid(row=0, column=1)
        Label(root, text="Annual Interest Rate", font="Cooper 15 bold", padx=20).grid(row=1, column=0, sticky=W)
        Label(root, text="Number of Years", font="Cooper 15 bold", padx=20, pady=10).grid(row=2, column=0, sticky=W)
        Label(root, text="Loan Amount", font="Cooper 15 bold", padx=20).grid(row=3, column=0, sticky=W)
        Label(root, text="Monthly Payment", font="Cooper 15 bold", padx=20, pady=10).grid(row=4, column=0, sticky=W)
        Label(root, text="Total Payment", font="Cooper 15 bold", padx=20).grid(row=5, column=0, sticky=W)

        # for taking inputs
        self.annualInterestRateVar = StringVar()
        self.numberOfYearsVar = StringVar()
        self.loanAmountVar = StringVar()
        self.monthlyPaymentVar = StringVar()
        self.totalPaymentVar = StringVar()

        # Entries for Label
        Entry(root, textvariable=self.annualInterestRateVar, justify=RIGHT, bd=5).grid(row=1, column=1, ipadx=5)
        Entry(root, textvariable=self.numberOfYearsVar, justify=RIGHT, bd=5).grid(row=2, column=1, pady=10, ipadx=5)
        Entry(root, textvariable=self.loanAmountVar, justify=RIGHT, bd=5).grid(row=3, column=1, ipadx=5)
        lblMonthlyPayment = Label(root, textvariable=self.monthlyPaymentVar).grid(row=4, column=1, pady=10, ipadx=5)
        lblTotalPayment = Label(root, textvariable=self.totalPaymentVar).grid(row=5, column=1, ipadx=5)

        # create the button
        btnComputePayment = Button(root, text="Compute Payment", command=self.computePayment, font="Cooper 15 bold", bg="grey", bd=5)
        btnComputePayment.grid(row=6, column=1, pady=10)

        root.mainloop()

    # Compute the monthly payment
    def getMonthlyPayment(self, loanAmount, monthlyInterestRate, numberOfYears):
        # compute the monthly payment.
        monthlyPayment = loanAmount * monthlyInterestRate / (1 - 1 / (1 + monthlyInterestRate) ** (numberOfYears * 12))
        return monthlyPayment

    # compute the total payment.
    def computePayment(self):
        monthlyPayment = self.getMonthlyPayment(float(self.loanAmountVar.get()), float(self.annualInterestRateVar.get())
                                                / 1200, int(self.numberOfYearsVar.get()))

        self.monthlyPaymentVar.set(format(monthlyPayment, '10.2f'))
        
        totalPayment = float(self.monthlyPaymentVar.get()) * 12 * int(self.numberOfYearsVar.get())

        self.totalPaymentVar.set(format(totalPayment, '10.2f'))


LoanCalculator()
