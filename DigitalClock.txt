from tkinter import * 
from time import strftime

root = Tk()
root.title("Digital Clock")


def time():
    string = strftime("%I : %M : %S : %p")
    l1.config(text=string)
    l1.after(1000, time)


l1 = Label(root, font="Cooper 50 bold", foreground="white", background="black")
l1.pack(anchor=CENTER)
time()

root.mainloop()
